int numbers = [5,4,3,2,1];
bool swap = true;

while (swap == true) {
    swap = false;
    for (int i = 0; i < numbers.length(); i++) {
        if (numbers[i] > numbers[i+1]) {
            swap = true;
            int temp = numbers[i];
            numbers[i] = numbers[i+1];
            numbers[i+1] = temp;
        }
    }
}
